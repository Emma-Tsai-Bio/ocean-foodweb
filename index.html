const handleIntroPointerUp = (e, type) => { if (interactionMode === 'move' && dragNode === type) { setDragNode(null); if (e.target.releasePointerCapture) e.target.releasePointerCapture(e.pointerId); e.stopPropagation(); } else if (interactionMode === 'draw' && introPhase === 1) { let targetType = type; // 針對 iPad 等觸控裝置：找出手指放開時，底下真實的卡片是誰 if (e.pointerType === 'touch' || e.pointerType === 'pen') { const elem = document.elementFromPoint(e.clientX, e.clientY); const cardElem = elem ? elem.closest('[data-card-type]') : null; if (cardElem) { targetType = cardElem.getAttribute('data-card-type'); } else { targetType = null; } } // 只要有起點、有合法的終點、且不是連到自己，就成功畫線！ if (drawingSource && targetType && drawingSource !== targetType && !edges.some(ed => ed.from === drawingSource && ed.to === targetType)) { setEdges([...edges, { from: drawingSource, to: targetType }]); setDrawingSource(null); // 連線成功，清空畫線狀態 } // 移除掉原本「沒連成功就強制清空」的邏輯，這樣手指放開後，A 卡片還會保持等待連線的狀態！ e.stopPropagation(); } };
